109        public void run() {
110                while(true){//ru-pu
111
112                        if(getIsAnimated()){
113                                this.perform();
114                        }
115                        //計算処理
116                        if(!getIsEdited()){
117                                this.reCalcPinionGear();
118                        }
119                        if(getIsRainbow()){
120                                this.rainbow();
121                        }
122                                //Viewに渡す
123                        this.changed();
124                        try {
125                                Thread.sleep(this.getTickTime());
126                        } catch (InterruptedException e) {
127                                e.printStackTrace();
128                        }
129                }
130
131        }

318        //rainbow処理
319        /**
320         * ペンの色をレインボーにするための処理を実行する。
321         */
322        public void rainbow(){
323
324                int r = 255;
325                int g = 0;
326                int b = 0;//Red Green Blue
327                //180度ごとに虹色にへんかするようにする
328                Double parTheta = this.getPinionGear().getTheta();
329                //初期位置の角度を取得
330                Double firstPinionTheta = this.getPinionGear().getPhi();
331                //初期位置からの位相ずれを取得
332                Double phaseShift = parTheta - firstPinionTheta;
333
334                phaseShift = phaseShift%Math.PI;
335
336                if(phaseShift>=0 && phaseShift<Math.PI/6){
337                        r = 255;
338                        g = 0 + (int)(255/(Math.PI/6) * (phaseShift));
339                        b = 0;
340                }
341                else if(phaseShift>=Math.PI/6 && phaseShift<Math.PI/3){
342                        r = 255 - (int)(255/(Math.PI/6) * (phaseShift -  Math.PI/6));
343                        g = 255;
344                        b = 0;
345                }
346                else if(phaseShift>=Math.PI/3 && phaseShift<Math.PI/2){
347                        r = 0;
348                        g = 255;
349                        b = 0 + (int)(255/(Math.PI/6) * (phaseShift -  Math.PI/3));
350                }
351                else if(phaseShift>=Math.PI/2 && phaseShift<Math.PI*2/3){
352                        r = 0;
353                        g = 255 - (int)(255/(Math.PI/6) * (phaseShift -  Math.PI/2));
354                        b = 255;
355                }
356                else if(phaseShift>=Math.PI*2/3 && phaseShift<Math.PI*5/6){
357                        r = 0 + (int)(255/(Math.PI/6) * (phaseShift -  Math.PI*2/3));
358                        g = 0;
359                        b = 255;
360                }
361                else if(phaseShift>=Math.PI*5/6 && phaseShift<Math.PI){
362                        r = 255;
363                        g = 0;
364                        b = 255 - (int)(255/(Math.PI/6) * (phaseShift -  Math.PI*5/6));
365                }
366
367                Color rainbowColor = new Color(r,g,b);
368
369                this.getPinionGear().setPenColor(rainbowColor);
370        }

418        /**
419         * 一つ前に描かれた絵を削除する
420         */
421        void spiroUndo(){
422                /*
423             * リスト内のnullを検索
424             * nullがないとき：->絵は一枚だけ->全削除
425             * nullがあるとき：複数枚の絵がある->最後のnullから最後までを削除
426                 *
427                 */
428                if(designLoci.size() != 0 ){
429                        int last = designLoci.size();
430                        int lastNull;
431
432                        if(designLoci.subList(0, last-1).contains(null)){
433                                lastNull = designLoci.subList(0,last-1).lastIndexOf(null);
434                }
435                else{
436                        lastNull = 0;
437                }
438                        designLoci.subList(lastNull,last).clear();
439                }
440                return;
441        }

516        /**
517         * ピニオンギアのパラメータを設定する
518         * @param aPoint2D マウスカーソルの座標
519         */
520        public void setPinionParameter(Point2D.Double aPoint2D)
521        {
522                //1 dの計算mouseとSpurの距離を計算
523                //2 rの計算
524                //3 φの計算piniondegreeのこと
525                //4 penDegreesも更新
526
527                getPinionGear().setCenter(aPoint2D);
528                //1
529                Double spd = getDistanceBetweenAToB(this.getSpurGear().getCenter(),aPoint2D);
530                getPinionGear().setPinionDistance(spd);
531
532                //2
533                Double pinionRadius;
534                if(this.isInscribe()){
535                        pinionRadius = this.getSpurGear().getRadius() - spd;
536
537                }else{
538                        pinionRadius = spd -this.getSpurGear().getRadius();
539                        //pinionの半径がspdより大きくなった時
540                        if(spd > this.getSpurGear().getRadius()*2-1){
541                                pinionRadius  = this.getSpurGear().getRadius();
542                        }
543
544                }
545                getPinionGear().setRadius(pinionRadius);
546
547                //3
548                getPinionGear().setTheta(this.getPinionDegrees());
549                pinionGear.setPhi(this.getPinionDegrees());
550
551        }

553        /**
554         * 座標上で二点間の距離を応答する
555         * @param aPointA 一つ目の座標
556         * @param aPointB 二つ目の座標
557         * @return 二点間の距離
558         */
559        public Double getDistanceBetweenAToB(Point2D.Double aPointA,Point2D.Double aPointB){
560                return Math.sqrt(Math.pow(aPointA.getX() - aPointB.getX(),2.0)
561                                                + Math.pow(aPointA.getY() - aPointB.getY(),2.0));
562        }

564        /**
565         * ピニオンギアの回転角を応答する
566         * @return ピニオンギアの回転角
567         */
568        public Double getPinionDegrees() { //pinionGearの初期位相ずれ
569                Double aPinionDegrees;
570
571                Point2D.Double aPinionCenter = this.getPinionGear().getCenter();
572                Point2D.Double aSpurCenter = this.getSpurGear().getCenter();
573
574                if(aPinionCenter.getX() > aSpurCenter.getX()) {
575                        aPinionDegrees = Math.atan((aSpurCenter.getY() - aPinionCenter.getY()) / (aSpurCenter.getX() - aPinionCenter.getX()));
576                } else {
577                        aPinionDegrees = Math.PI + Math.atan((aSpurCenter.getY() - aPinionCenter.getY()) / (aSpurCenter.getX() - aPinionCenter.getX()));
578                }
579                return aPinionDegrees;
580        }

620        /**
621         * ピニオンペンの角度を応答する
622         * @param aPoint2D マウスカーソルの座標
623         * @return ピニオンペンの角度
624         */
625        public Double calcPenDegrees(Point2D.Double aPoint2D){
626                //外接時の計算
627                if(!isInscribe()){
628                        if(aPoint2D.getX() > getPinionGear().getCenter().getX()){
629                                return Math.PI + Math.atan( (aPoint2D.getY() -  getPinionGear().getCenter().getY())/(aPoint2D.getX() -  getPinionGear().getCenter().getX() ));
630                        }
631                        else{
632                                return  Math.atan( (aPoint2D.getY() -  getPinionGear().getCenter().getY())/(aPoint2D.getX() -  getPinionGear().getCenter().getX() ));
633                        }
634                }
635                //内接時の計算
636                else{
637                        if(aPoint2D.getX() > getPinionGear().getCenter().getX()){
638                                return Math.atan( (aPoint2D.getY() -  getPinionGear().getCenter().getY())/(aPoint2D.getX() -  getPinionGear().getCenter().getX() ));
639                        }
640                        else{
641                                return Math.PI + Math.atan( (aPoint2D.getY() -  getPinionGear().getCenter().getY())/(aPoint2D.getX() -  getPinionGear().getCenter().getX() ));
642                        }
643                }
644        }

646        /**
647         * 変更後のピニオンギア状態から軌跡の数式のパラメータを求める
648         */
649        public void reCalcPinionGear(){
650
651                PinionGear pinionGear = this.getPinionGear();
652                SpurGear spurGear = this.getSpurGear();
653
654                Double sx = spurGear.getCenter().getX();
655                Double sy = spurGear.getCenter().getY();
656                Double sr = spurGear.getRadius();
657                Double pr = pinionGear.getRadius();
658
659
660                Double px,py;
661                if(pinionGear.getPinionDistance() < spurGear.getRadius()){
662                        px = sx + (sr - pr) * Math.cos(this.getPinionGear().getTheta());
663                        py = sy + (sr - pr) * Math.sin(this.getPinionGear().getTheta());
664                }
665                else{
666                        px = sx + (sr + pr) * Math.cos(this.getPinionGear().getTheta());
667                        py = sx + (sr + pr) * Math.sin(this.getPinionGear().getTheta());
668                }
669                this.getPinionGear().setCenter(new Point2D.Double(px, py));
670
671                if(getPinionGear().getPinionDistance() < getSpurGear().getRadius()){
672                        pinionDegree = -getPinionGear().getTheta() *(getSpurGear().getRadius() - getPinionGear().getRadius())/getPinionGear().getRadius();
673                        getPinionGear().setPinionDegrees(pinionDegree);
674                }
675                else{
676                        pinionDegree = getPinionGear().getTheta() *(getSpurGear().getRadius() + getPinionGear().getRadius())/getPinionGear().getRadius();
677                        getPinionGear().setPinionDegrees(pinionDegree);
678                }
679
680                //pen
